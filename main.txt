mode="menu" --default game state
function waitForInput()
	if mode=="menu" then
		cls(2)
		print("MENU", 50, 50, 12)
	end
	if key(24) then --x
		mode = "game"
	end
end

function menu()
		if mode=="game" then
			cls(1)
			print("welcome to the game user!", 10, 10, 12)
		end
		if key(26) then
			mode="menu"
		end
end
